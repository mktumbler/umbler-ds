## Uso

```tsx
import { Alert } from '@/components/ui/alert';

// Simples
<Alert variant="success">Deploy concluído com sucesso.</Alert>

// Com título e dismiss
<Alert
  variant="warning"
  title="Domínio expirando"
  onClose={() => setVisible(false)}
>
  Seu domínio <strong>umbler.com</strong> expira em 7 dias.
</Alert>
```
